  document.querySelectorAll('.mode-tab').forEach(b => b.classList.remove('active'));

function autoResize(el) {

  el.style.height = 'auto';

  el.style.height = Math.min(el.scrollHeight, 160) + 'px';

}

  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); }

function formatMarkdown(text) {

  let html = text

    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')

    // Headings

    .replace(/^## (.+)$/gm, '<h3>$1</h3>')

    .replace(/^#### (.+)$/gm, '<h4>$1</h4>')

    .replace(/^### (.+)$/gm, '<h3>$1</h3>')

    // Bold

    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')

    // Code

    .replace(/`([^`]+)`/g, '<code>$1</code>')

    // Lists

    .replace(/^- (.+)$/gm, '<li>$1</li>')

    .replace(/^\d+\. (.+)$/gm, '<li>$1</li>');


  // Wrap consecutive <li> in <ul>

  html = html.replace(/(<li>[\s\S]*?<\/li>)(\n<li>[\s\S]*?<\/li>)*/g, m => `<ul>${m}</ul>`);


  // Paragraphs for non-block lines

  const lines = html.split('\n');

  const out = [];

  let i = 0;

  while (i < lines.length) {

    const line = lines[i].trim();

    if (!line) { i++; continue; }

    if (line.startsWith('<h') || line.startsWith('<ul') || line.startsWith('<li')) {

      out.push(line);

    } else {

      // Collect consecutive non-block lines into a paragraph

      let para = line;

      while (i + 1 < lines.length) {

        const next = lines[i+1].trim();

        if (!next || next.startsWith('<h') || next.startsWith('<ul') || next.startsWith('<li')) break;

        para += ' ' + next;

        i++;

      }

      out.push(`<p>${para}</p>`);

    }

    i++;

  }

  return out.join('\n');

}


async function sendMessage() {

  if (isThinking) return;

  const input = document.getElementById('userInput');

  const text = input.value.trim();

  if (!text) return;


  const modeTag = { brief: '[BRIEF]', roadmap: '[ROADMAP]', deep: '[DEEP DIVE]', tools: '[TOOLS & METHODS]' }[currentMode];


  addUserMessage(text);

  input.value = '';

  input.style.height = 'auto';

  conversationHistory.push({ role: 'user', content: `${modeTag} ${text}` });


  isThinking = true;

  document.getElementById('sendBtn').disabled = true;

  addThinking();


  try {

    const response = await fetch('https://api.anthropic.com/v1/messages', {

      method: 'POST',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify({

        model: 'claude-sonnet-4-20250514',

        max_tokens: 1000,

        system: SYSTEM_PROMPT,

        messages: conversationHistory

      })

    });


    const data = await response.json();

    const reply = data.content?.[0]?.text || 'I was unable to generate a response. Please try again.';


    removeThinking();

    addAdvisorMessage(reply);

    conversationHistory.push({ role: 'assistant', content: reply });


  } catch (err) {

    removeThinking();

    addAdvisorMessage(`**Error:** Unable to connect. Please check your connection and try again.\n\n\`${err.message}\``);

  }


  isThinking = false;

  document.getElementById('sendBtn').disabled = false;

}
